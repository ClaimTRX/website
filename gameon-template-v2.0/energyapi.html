Rent Energy with user.js

The user.js script lets you delegate energy on the TRON blockchain. You can pay either directly with TRX (sent automatically from your wallet) or using credits. Pricing: 65,000 energy units for 6.5 TRX, or 135,000 units for 13 TRX (5-minute duration).

To use, copy the script below, install Node.js and the tronweb package (npm install tronweb), update the configuration variables, and run with node user.js.

user.js Code

const TronWeb = require('tronweb').TronWeb;

// === CONFIGURATION ===
const TRONGRID_API_KEY = "YOUR-TRONGRID-API-KEY";
const PRIVATE_KEY = "YOUR-PRIVATE-KEY";
const API_KEY = "YOU-API-KEY";
const USE_CREDITS = false;
const PAYMENT_WALLET = "TRUnBRHsGVYeFuBccYac5wyWYBAgcnLzmn"; // DO NOT CHANGE

// === ORDER SETUP ===
const order = {
    receiver: "RECEVING WALLET", // Receiving wallet
    resource_type: "energy",
    amount: 65000,  // Options: 65000 or 135000
    duration: 5     // Duration in minutes (DO NOT CHANGE)
};

// === PRICE LOOKUP ===
const PRICE_TABLE = {
    65000: 6.5,
    135000: 13
};
const trxPrice = PRICE_TABLE[order.amount];
if (!trxPrice) {
    console.error("❌ Invalid order amount. Must be 65000 or 135000.");
    process.exit(1);
}
const priceInSun = trxPrice * 1e6;

// === TRON SETUP ===
const tronWeb = new TronWeb({
    fullHost: 'https://api.trongrid.io',
    headers: { 'TRON-PRO-API-KEY': TRONGRID_API_KEY }
});
tronWeb.setPrivateKey(PRIVATE_KEY);
const senderAddress = tronWeb.address.fromPrivateKey(PRIVATE_KEY);

// === API SETUP ===
const url = "http://api2.cftecosystem.com/energy";
const headers = {
    "Content-Type": "application/json",
    "Api-Key": API_KEY
};

// === MAIN FUNCTION ===
(async () => {
    console.log(`🔑 Sender: ${senderAddress}`);
    console.log(`💸 Paying ${trxPrice} TRX for ${order.amount} energy (Duration: ${order.duration} min)`);
    console.log(`➡️ To vendor: ${PAYMENT_WALLET}`);
    console.log(`🎯 Energy receiver: ${order.receiver}`);

    if (USE_CREDITS) {
        const res = await fetch(url + "/create-order", {
            method: "POST",
            headers,
            body: JSON.stringify({ order, use_credits: true })
        });
        const result = await res.json();
        console.log(result);
        return;
    }

    try {
        // === Build & Sign TX ===
        const tx = await tronWeb.transactionBuilder.sendTrx(
            PAYMENT_WALLET,
            priceInSun,
            senderAddress
        );
        tx.feeLimit = 1_000_000;

        const signedTx = await tronWeb.trx.sign(tx, PRIVATE_KEY);

        // === Broadcast TX ===
        const broadcastRes = await tronWeb.trx.sendRawTransaction(signedTx);
        console.log("📡 Broadcast response:", broadcastRes);

        if (!broadcastRes.result) {
            console.error("❌ Failed to broadcast transaction.");
            return;
        }

        console.log(`✅ TX broadcasted: ${signedTx.txID}`);

        // === Call /create-order with full signed TX ===
        const res = await fetch(url + "/create-order", {
            method: "POST",
            headers,
            body: JSON.stringify({
                order,
                use_credits: false,
                signed_tx: signedTx // ✅ required by backend
            })
        });

        const text = await res.text();
        let data;
        try {
            data = JSON.parse(text);
        } catch (err) {
            console.error("❌ Failed to parse response:");
            console.log(text);
            return;
        }

        if (!res.ok || data.success === false) {
            console.error("❌ Order creation failed:", data);
            return;
        }

        console.log("✅ Order created:", data);

        // === Poll for Delegation Status ===
        const orderId = data.data.order_id;
        const statusUrl = `${url}/delegation-status?requestId=${orderId}`;
        let attempts = 0;

        const interval = setInterval(async () => {
            attempts++;
            try {
                const statusRes = await fetch(statusUrl, { method: 'GET', headers });
                const statusData = await statusRes.json();
                console.log(`ℹ️ Status [${attempts}]:`, statusData);

                if (statusData.status === "delegated") {
                    console.log(`✅ Energy delegated: TxID ${statusData.txId}`);
                    clearInterval(interval);
                } else if (statusData.status === "failed" || attempts >= 60) {
                    console.log(`❌ Delegation failed or timed out.`);
                    clearInterval(interval);
                }
            } catch (err) {
                console.error(`⚠️ Polling error: ${err.message}`);
                if (attempts >= 60) clearInterval(interval);
            }
        }, 2000);

    } catch (err) {
        console.error("❌ Transaction error:", err.message);
    }
})();
